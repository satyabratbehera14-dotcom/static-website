// ASP.NET Core 8 Minimal API — Program.cs
var builder = WebApplication.CreateBuilder(args);
builder.Services.AddCors(opt => opt.AddPolicy("AllowAll",
    p => p.AllowAnyOrigin().AllowAnyMethod().AllowAnyHeader()));
builder.Services.AddSingleton<YoutubeDownloadService>();

var app = builder.Build();
app.UseCors("AllowAll");

// GET /api/info?url=...
app.MapGet("/api/info", async (
    string url, YoutubeDownloadService svc) => {
    var info = await svc.GetVideoInfoAsync(url);
    return info is null
        ? Results.BadRequest("Could not fetch video info")
        : Results.Ok(info);
});

// POST /api/download
app.MapPost("/api/download", async (
    HttpContext ctx, YoutubeDownloadService svc) => {
    var req = await JsonSerializer
        .DeserializeAsync<DownloadRequest>(ctx.Request.Body);
    if (req is null) return Results.BadRequest();
    var path = await svc.DownloadAsync(req);
    if (path is null) return Results.Problem("Download failed");
    var mime = req.Format == "mp3" ? "audio/mpeg" : "video/mp4";
    return Results.File(path, mime, Path.GetFileName(path));
});
app.Run();
public class YoutubeDownloadService {
  private const string YtDlp = "yt-dlp";
  private readonly string _outputDir;

  public YoutubeDownloadService() {
    _outputDir = Path.Combine(
        Path.GetTempPath(), "yt-downloads");
    Directory.CreateDirectory(_outputDir);
  }

  public async Task<VideoInfo?> GetVideoInfoAsync(string url) {
    var args = $"--dump-single-json --no-playlist \"{url}\"";
    var json = await RunProcessAsync(YtDlp, args);
    if (json is null) return null;
    using var doc = JsonDocument.Parse(json);
    var r = doc.RootElement;
    return new VideoInfo {
        Title    = r.GetProperty("title").GetString() ?? "",
        Author   = r.GetProperty("uploader").GetString() ?? "",
        Duration = FormatDuration(
                     r.GetProperty("duration").GetInt32()),
        Thumbnail= r.GetProperty("thumbnail").GetString() ?? "",
    };
  }

  public async Task<string?> DownloadAsync(DownloadRequest req) {
    var id  = Guid.NewGuid().ToString();
    var tpl = Path.Combine(_outputDir, $"{id}.%(ext)s");
    string args;
    if (req.Format == "mp3")
        args = $"-x --audio-format mp3 -o \"{tpl}\" \"{req.Url}\"";
    else {
        var q = req.Quality switch {
            "1080p" => "bestvideo[height<=1080]+bestaudio",
            "720p"  => "bestvideo[height<=720]+bestaudio",
            "480p"  => "bestvideo[height<=480]+bestaudio",
            _       => "bestvideo+bestaudio"
        };
        args = $"-f \"{q}\" --merge-output-format mp4 -o \"{tpl}\" \"{req.Url}\"";
    }
    await RunProcessAsync(YtDlp, args);
    var ext  = req.Format == "mp3" ? "mp3" : "mp4";
    var file = Path.Combine(_outputDir, $"{id}.{ext}");
    return File.Exists(file) ? file : null;
  }
}
# 1. Install .NET 8 SDK
#    https://dotnet.microsoft.com/download

# 2. Install yt-dlp (required backend dependency)
#    Windows
winget install yt-dlp

#    macOS
brew install yt-dlp

#    Linux
sudo curl -L https://github.com/yt-dlp/yt-dlp/releases/latest\
  /download/yt-dlp -o /usr/local/bin/yt-dlp
sudo chmod a+rx /usr/local/bin/yt-dlp

# 3. Install ffmpeg (for audio extraction & muxing)
#    Windows: winget install Gyan.FFmpeg
#    macOS:   brew install ffmpeg
#    Linux:   sudo apt install ffmpeg

# 4. Create & run the API
dotnet new web -n YouTubeDownloader.Api
# Copy Program.cs into the project
cd YouTubeDownloader.Api
dotnet run

# API listens on http://localhost:5000
# GET  /api/info?url=<youtube-url>
# POST /api/download  { url, format, quality }
